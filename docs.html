01 Introduction & Philosophy

HOLEXA is a modern, open-source programming language designed from the ground up to be fast, safe, readable, and simple. It combines the best ideas from the world's most successful languages into one cohesive, powerful tool for developers of all levels.

HOLEXA was built with one core belief: programming should be powerful without being painful. Every feature exists to help you write better code faster, with fewer bugs and more clarity.

Design Goals

⚡
Fast like C
Compiled to native code. Zero overhead runtime. Performance-critical applications run at full speed.
🛡️
Safe like Rust
Memory safety guaranteed. No null pointer crashes, no buffer overflows, no undefined behavior.
📖
Readable like Python
Clean, expressive syntax. Code reads like English. Easy to learn, easy to maintain.
🔧
Simple like Go
Minimal complexity. One obvious way to do things. Fast compilation. Clear error messages.
🌐
Cross Platform
Linux, macOS, Windows, iPad. Write once, run everywhere with no changes.
🐉
Self-Hosting
The HOLEXA compiler is written in HOLEXA itself — a mark of a mature, complete language.

Who is HOLEXA for?

HOLEXA is designed for developers who want a language that is both powerful and approachable. Whether you are a beginner learning programming for the first time, or an experienced developer building production systems, HOLEXA scales with your needs.

💡
If you know Python, you already know most of HOLEXA's syntax. The transition is smooth and natural.

HOLEXA is statically typed with full type inference. You declare variables with let (mutable) or const (immutable).

variables.hlx
fn main() {
    let x     = 42          // Int
    let name  = "Dragon"    // String
    let pi    = 3.14159    // Float
    let flag  = true        // Bool
    const MAX = 100         // Cannot reassign

    // Reassignment (only for let)
    x = x + 1             // x is now 43

    // String concatenation with +
    let msg = "Hello " + name + "!"
    print(msg)             // Hello Dragon!

    // Operators
    print(str(10 + 5))    // 15
    print(str(10 - 3))    // 7
    print(str(4 * 5))     // 20
    print(str(10 / 3))    // 3 (integer)
    print(str(10 % 3))    // 1 (remainder)
    print(str(2 ** 10))   // 1024 (power)
}

07 Functions

Functions are defined with the fn keyword. They support type annotations, return types, multiple parameters, and full recursion.

functions.hlx
// Basic function — returns a String
fn greet(name: String) -> String {
    return "Hello " + name + "!"
}

// Multiple parameters
fn add(a: Int, b: Int) -> Int {
    return a + b
}

// Boolean return value
fn is_even(n: Int) -> Bool {
    if n % 2 == 0 { return true }
    return false
}

// No return value (void)
fn log(msg: String) {
    print("[LOG] " + msg)
}

// Recursive function
fn factorial(n: Int) -> Int {
    if n <= 1 { return 1 }
    return n * factorial(n - 1)
}

fn main() {
    print(greet("Dragon"))           // Hello Dragon!
    print(str(add(10, 20)))          // 30
    print(str(is_even(4)))          // true
    print(str(is_even(7)))          // false
    log("Program started")           // [LOG] Program started
    print(str(factorial(5)))        // 120
}

08 Control Flow

HOLEXA supports complete conditional control with if, else if, and else blocks.

control.hlx
fn grade(score: Int) -> String {
    if score >= 90 {
        return "A+"
    } else if score >= 80 {
        return "A"
    } else if score >= 70 {
        return "B"
    } else if score >= 60 {
        return "C"
    } else {
        return "F"
    }
}

fn main() {
    print(grade(95))   // A+
    print(grade(75))   // B
    print(grade(45))   // F

    // Logical operators
    let x = 42
    if x > 10 and x < 100 {
        print("x is between 10 and 100")
    }

    if x < 0 or x > 1000 {
        print("x is out of range")
    } else {
        print("x is in range")
    }

    if not (x == 0) {
        print("x is not zero")
    }
}

09 Loops

for Loop (Range-based)

loops.hlx
fn main() {
    // Basic for loop — 0 to 4 (end is exclusive)
    for i in 0..5 {
        print("i = " + str(i))
    }

    // Sum 1 to 100
    let sum = 0
    for i in 1..101 {
        sum = sum + i
    }
    print("Sum = " + str(sum))   // 5050

    // Nested loops — multiplication table
    for i in 1..4 {
        for j in 1..4 {
            print(str(i) + " x " + str(j) + " = " + str(i * j))
        }
    }

    // while loop
    let x = 0
    while x < 5 {
        print("x = " + str(x))
        x = x + 1
    }

    // Countdown
    let n = 5
    while n > 0 {
        print(str(n) + "...")
        n = n - 1
    }
    print("Blastoff!")
}

10 Classes & Objects

HOLEXA supports full object-oriented programming with classes, fields, methods, and the self keyword.

classes.hlx
class Dragon {
    name:   String
    power:  Int
    alive:  Bool

    // Constructor method
    fn new(name: String, power: Int) -> Dragon {
        return Dragon {
            name:  name,
            power: power,
            alive: true
        }
    }

    // Method using self to access fields
    fn roar(self) -> String {
        return self.name + " ROARS with power " + str(self.power) + "!"
    }

    fn is_legendary(self) -> Bool {
        return self.power > 9000
    }

    fn describe(self) -> String {
        return "Dragon[" + self.name + ", power=" + str(self.power) + "]"
    }
}

fn main() {
    let d = Dragon.new("HOLEXA", 9999)
    print(d.roar())                       // HOLEXA ROARS with power 9999!
    print(d.describe())                   // Dragon[HOLEXA, power=9999]
    print(str(d.is_legendary()))           // true

    let d2 = Dragon.new("Fire", 5000)
    print(d2.roar())                      // Fire ROARS with power 5000!
}

11 Error Handling

HOLEXA uses try/catch for safe error handling. Programs never crash silently — all errors are catchable.

errors.hlx
fn divide(a: Int, b: Int) -> Int {
    if b == 0 {
        print("[ERROR] Division by zero!")
        return 0
    }
    return a / b
}

fn main() {
    // try/catch block
    try {
        print(str(divide(10, 2)))   // 5
        print(str(divide(10, 0)))   // [ERROR] + 0
        print("Code continues!")
    } catch err {
        print("Caught: " + err)
    }

    // Nested try/catch
    try {
        try {
            print("inner try")
        } catch inner_err {
            print("inner catch: " + inner_err)
        }
        print("outer try continues")
    } catch outer_err {
        print("outer catch: " + outer_err)
    }

    print("Program completed!")
}

13 Recursion

HOLEXA fully supports recursive functions. No special syntax needed — just call the function within itself.

recursion.hlx
fn factorial(n: Int) -> Int {
    if n <= 1 { return 1 }
    return n * factorial(n - 1)
}

fn fibonacci(n: Int) -> Int {
    if n <= 1 { return n }
    return fibonacci(n - 1) + fibonacci(n - 2)
}

fn power(base: Int, exp: Int) -> Int {
    if exp == 0 { return 1 }
    return base * power(base, exp - 1)
}

fn main() {
    print("Factorials:")
    for i in 1..8 {
        print(str(i) + "! = " + str(factorial(i)))
    }
    print("Fibonacci:")
    for i in 0..10 {
        print("fib(" + str(i) + ") = " + str(fibonacci(i)))
    }
    print("Powers: 2^10 = " + str(power(2, 10)))
}

21 Integration Guide

HOLEXA works alongside other programming languages through shell integration and process communication.

HOLEXA with Python

bridge.py
import subprocess, tempfile, os

def run_holexa(code: str) -> str:
    """Run HOLEXA code from Python and return output"""
    with tempfile.NamedTemporaryFile(suffix='.hlx', mode='w', delete=False) as f:
        f.write(code)
        fname = f.name
    try:
        result = subprocess.run(
            ['hlxc', 'run', fname],
            capture_output=True, text=True, timeout=10
        )
        return result.stdout
    finally:
        os.unlink(fname)

output = run_holexa('''
fn main() {
    for i in 0..10 {
        print(str(i * i))
    }
}
''')
print("HOLEXA output:", output)

HOLEXA with JavaScript / Node.js

bridge.js
const { execSync } = require('child_process');
const fs = require('fs');

function runHolexa(code) {
    fs.writeFileSync('/tmp/temp.hlx', code);
    try {
        return execSync('hlxc run /tmp/temp.hlx').toString();
    } catch(e) {
        return e.stderr.toString();
    }
}

const output = runHolexa(`
fn main() {
    for i in 1..6 {
        print(str(i) + "! = " + str(i))
    }
}
`);
console.log(output);

22 Advanced Examples

Prime Number Sieve

primes.hlx
fn is_prime(n: Int) -> Bool {
    if n < 2 { return false }
    if n == 2 { return true }
    if n % 2 == 0 { return false }
    let i = 3
    while i * i <= n {
        if n % i == 0 { return false }
        i = i + 2
    }
    return true
}

fn count_primes(limit: Int) -> Int {
    let count = 0
    for i in 2..(limit + 1) {
        if is_prime(i) {
            count = count + 1
        }
    }
    return count
}

fn main() {
    print("Prime numbers up to 50:")
    for i in 2..51 {
        if is_prime(i) {
            print("  " + str(i))
        }
    }
    print("Total primes up to 100: " + str(count_primes(100)))
}

Scientific Calculator

calculator.hlx
fn calculate(a: Int, op: String, b: Int) -> Int {
    if op == "+" { return a + b }
    if op == "-" { return a - b }
    if op == "*" { return a * b }
    if op == "/" {
        if b == 0 {
            print("[ERROR] Division by zero!")
            return 0
        }
        return a / b
    }
    if op == "%" { return a % b }
    print("[ERROR] Unknown operator: " + op)
    return 0
}

fn main() {
    print("Calculator Demo:")
    print("10 + 5  = " + str(calculate(10, "+", 5)))
    print("10 - 3  = " + str(calculate(10, "-", 3)))
    print("4 * 5   = " + str(calculate(4,  "*", 5)))
    print("10 / 3  = " + str(calculate(10, "/", 3)))
    print("10 % 3  = " + str(calculate(10, "%", 3)))
    print("10 / 0  = " + str(calculate(10, "/", 0)))
}
