Install HOLEXA on any platform. After installation you can write .hlx files and run them instantly with hlxc run.

Language/Variables

Variables & Types

HOLEXA is statically typed with full type inference. You rarely need to write types explicitly.

TypeExampleDescription
Intlet x = 42Integer number (64-bit)
Floatlet f = 3.14Floating point (64-bit)
Stringlet s = "hello"UTF-8 text
Boollet b = truetrue or false
Nonelet n = noneNo value (null equivalent)
variables.hlx
fn main() {
    let x     = 42
    let name  = "Dragon"
    let pi    = 3.14159
    let flag  = true
    const MAX = 100

    // Reassignment
    x = x + 1

    // String concatenation
    let msg = "Hello " + name + "!"
    print(msg)

    // Type conversion
    print(str(x))
    print(str(pi))
    print(str(flag))
}
ℹ️
const variables cannot be reassigned after declaration. Use let for mutable variables.
Language/Functions

Functions

Functions are first-class in HOLEXA. They support type annotations, return types, and full recursion.

functions.hlx
// Basic function
fn greet(name: String) -> String {
    return "Hello " + name
}

// Multiple parameters
fn add(a: Int, b: Int) -> Int {
    return a + b
}

// Boolean return value
fn is_even(n: Int) -> Bool {
    if n % 2 == 0 { return true }
    return false
}

// Recursive function
fn factorial(n: Int) -> Int {
    if n <= 1 { return 1 }
    return n * factorial(n - 1)
}

fn main() {
    print(greet("Dragon"))
    print(str(add(10, 20)))
    print(str(is_even(4)))
    print(str(factorial(5)))
}

Control Flow

fn grade(score: Int) -> String {
    if score >= 90 {
        return "A+"
    } else if score >= 80 {
        return "A"
    } else if score >= 70 {
        return "B"
    } else if score >= 60 {
        return "C"
    } else {
        return "F"
    }
}

fn main() {
    print(grade(95))
    print(grade(75))
    print(grade(45))

    let x = 42
    if x > 10 and x < 100 {
        print("x is between 10 and 100")
    }
}
Language/Loops

Loops

for loops with ranges, while loops, and nested loops.

loops.hlx
fn main() {
    // for loop with range (0 to 4)
    for i in 0..5 {
        print("i = " + str(i))
    }

    // while loop
    let x = 0
    while x < 5 {
        print("x = " + str(x))
        x = x + 1
    }

    // Sum using for loop
    let sum = 0
    for i in 1..101 {
        sum = sum + i
    }
    print("Sum 1-100 = " + str(sum))

    // Nested loops
    for i in 1..4 {
        for j in 1..4 {
            print(str(i) + " x " + str(j) + " = " + str(i * j))
        }
    }
}
Language/Classes

Classes

Object-oriented programming with classes, fields, methods, and the self keyword.

classes.hlx
class Dragon {
    name:  String
    power: Int
    alive: Bool

    fn new(name: String, power: Int) -> Dragon {
        return Dragon {
            name: name,
            power: power,
            alive: true
        }
    }

    fn roar(self) -> String {
        return self.name + " ROARS with power " + str(self.power) + "!"
    }

    fn is_legendary(self) -> Bool {
        return self.power > 9000
    }

    fn describe(self) -> String {
        return "Dragon[" + self.name + ", power=" + str(self.power) + "]"
    }
}

fn main() {
    let d = Dragon.new("HOLEXA", 9999)
    print(d.roar())
    print(d.describe())
    print("Legendary: " + str(d.is_legendary()))
}
Language/Error Handling

Error Handling

HOLEXA uses try/catch for safe error handling. Errors never crash silently.

errors.hlx
fn divide(a: Int, b: Int) -> Int {
    if b == 0 {
        print("[ERROR] Cannot divide by zero!")
        return 0
    }
    return a / b
}

fn safe_sqrt(n: Int) -> Int {
    if n < 0 {
        print("[ERROR] Cannot sqrt negative number")
        return 0
    }
    return sqrt(n)
}

fn main() {
    // try/catch block
    try {
        print("10 / 2 = " + str(divide(10, 2)))
        print("10 / 0 = " + str(divide(10, 0)))
        print("After error — code continues!")
    } catch err {
        print("Caught error: " + err)
    }

    print("Program completed successfully")
}
ℹ️
Best Practice — Always validate inputs before operations. Use try/catch for external operations like file I/O and network calls.

Language/Recursion

Recursion

HOLEXA fully supports recursive functions. No stack depth limits for reasonable programs.

recursion.hlx
// Factorial
fn factorial(n: Int) -> Int {
    if n <= 1 { return 1 }
    return n * factorial(n - 1)
}

// Fibonacci
fn fibonacci(n: Int) -> Int {
    if n <= 1 { return n }
    return fibonacci(n - 1) + fibonacci(n - 2)
}

// Power function
fn power(base: Int, exp: Int) -> Int {
    if exp == 0 { return 1 }
    return base * power(base, exp - 1)
}

fn main() {
    print("Factorials:")
    for i in 1..8 {
        print(str(i) + "! = " + str(factorial(i)))
    }

    print("Fibonacci:")
    for i in 0..10 {
        print("fib(" + str(i) + ") = " + str(fibonacci(i)))
    }

    print("Powers of 2:")
    for i in 0..10 {
        print("2^" + str(i) + " = " + str(power(2, i)))
    }
}

Stdlib/fs

fs

File system operations. Import with import fs.

fs_example.hlx
import fs

fn main() {
    // Write a file
    fs.write("data.txt", "Hello HOLEXA!\nLine 2\nLine 3")

    // Read a file
    let content = fs.read("data.txt")
    print(content)

    // Check existence
    let exists = fs.exists("data.txt")
    print("Exists: " + str(exists))

    // Delete file
    fs.delete("data.txt")
    print("Deleted!")
}
